35  package org.owasp.encoder;
36  
37  import junit.framework.Test;
38  import junit.framework.TestCase;
39  import junit.framework.TestSuite;
40  
41  /**
42   * XMLEncoderTest -- test suite for the XMLEncoder.
43   *
44   * @author Jeff Ichnowski
45   */
46  public class XMLEncoderTest extends TestCase {
47      public static Test suite() {
48          TestSuite suite = new TestSuite();
49          for (XMLEncoder.Mode mode : XMLEncoder.Mode.values()) {
50              EncoderTestSuiteBuilder builder = new EncoderTestSuiteBuilder(new XMLEncoder(mode), "-safe-", "-&-")
51                  .encode("safe", "safe")
52                  .encode("unencoded &amp; encoded", "unencoded & encoded")
53                  .encode("invalid-control-characters", " b ", "\0b\26")
54                  .encode("valid-surrogate-pair", "\ud800\udc00", "\ud800\udc00")
55                  .encode("missing-low-surrogate", " ", "\ud800")
56                  .encode("missing-high-surrogate", " ", "\udc00")
57                  .encode("valid-upper-char", "\ufffd", "\ufffd")
58                  .encode("invalid-upper-char", " ", "\uffff")
59  
60                  .invalid(0, 0x1f)
61                  .valid("\t\r\n")
62                  .valid(' ', Character.MAX_CODE_POINT)
63                  .invalid(0x7f, 0x9f)
64                  .valid("\u0085")
65                  .invalid(Character.MIN_SURROGATE, Character.MAX_SURROGATE)
66                  .invalid(0xfdd0, 0xfdef)
67                  .invalid(0xfffe, 0xffff)
68                  .invalid(0x1fffe, 0x1ffff)
69                  .invalid(0x2fffe, 0x2ffff)
70                  .invalid(0x3fffe, 0x3ffff)
71                  .invalid(0x4fffe, 0x4ffff)
72                  .invalid(0x5fffe, 0x5ffff)
73                  .invalid(0x6fffe, 0x6ffff)
74                  .invalid(0x7fffe, 0x7ffff)
75                  .invalid(0x8fffe, 0x8ffff)
76                  .invalid(0x9fffe, 0x9ffff)
77                  .invalid(0xafffe, 0xaffff)
78                  .invalid(0xbfffe, 0xbffff)
79                  .invalid(0xcfffe, 0xcffff)
80                  .invalid(0xdfffe, 0xdffff)
81                  .invalid(0xefffe, 0xeffff)
82                  .invalid(0xffffe, 0xfffff)
83                  .invalid(0x10fffe, 0x10ffff);
84  
85              switch (mode) {
86              case ALL:
87                  builder.encoded("&><\'\"")
88                      .encode("&amp;", "&")
89                      .encode("&gt;", ">")
90                      .encode("&lt;", "<")
91                      .encode("&#39;", "\'")
92                      .encode("&#34;", "\"");
93                  break;
94              case CONTENT:
95                  builder.encoded("&><")
96                      .encode("&amp;", "&")
97                      .encode("&gt;", ">")
98                      .encode("&lt;", "<")
99                      .encode("\'", "\'")
100                     .encode("\"", "\"");
101                 break;
102             case ATTRIBUTE:
103                 builder.encoded("&<\'\"")
104                     .encode("&amp;", "&")
105                     .encode(">", ">")
106                     .encode("&lt;", "<")
107                     .encode("&#39;", "\'")
108                     .encode("&#34;", "\"");
109                 break;
110             case SINGLE_QUOTED_ATTRIBUTE:
111                 builder.encoded("&<\'")
112                     .encode("&amp;", "&")
113                     .encode(">", ">")
114                     .encode("&lt;", "<")
115                     .encode("&#39;", "\'")
116                     .encode("\"", "\"");
117                 break;
118             case DOUBLE_QUOTED_ATTRIBUTE:
119                 builder.encoded("&<\"")
120                     .encode("&amp;", "&")
121                     .encode(">", ">")
122                     .encode("&lt;", "<")
123                     .encode("\'", "\'")
124                     .encode("&#34;", "\"");
125                 break;
126             default:
127                 throw new AssertionError("untested mode: "+mode);
128             }
129 
130             suite.addTest(builder
131                 .validSuite()
132                 .invalidSuite(XMLEncoder.INVALID_CHARACTER_REPLACEMENT)
133                 .encodedSuite()
134                 .build());
135         }
136         return suite;
137     }
138 }
