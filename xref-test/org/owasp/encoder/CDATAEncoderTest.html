35  package org.owasp.encoder;
36  
37  import junit.framework.Test;
38  import junit.framework.TestCase;
39  
40  /**
41   * CDATAEncoderTest -- Test the CDATAEncoder for all code-points.
42   *
43   * @author Jeff Ichnowski
44   */
45  public class CDATAEncoderTest extends TestCase {
46      public static Test suite() {
47          return new EncoderTestSuiteBuilder(CDATAEncoderTest.class, new CDATAEncoder(), "-safe-", "-]]>-")
48              .encode("]]]]><![CDATA[>", "]]>")
49              .encode("]", "]")
50              .encode("]]", "]]")
51              .encode("]]]]><![CDATA[>]", "]]>]")
52              .encode("]]]]><![CDATA[>]>", "]]>]>")
53              .encode("]]]]><![CDATA[>>", "]]>>")
54              .encode("]]]]]", "]]]]]")
55              .encode("<\"&\'>", "<\"&\'>") // valid in CDATA, not in XML
56  
57              .invalid(0, 0x1f)
58              .valid("\t\r\n")
59              .valid(' ', Character.MAX_CODE_POINT)
60              .invalid(0x7f, 0x9f)
61              .valid("\u0085")
62              .invalid(Character.MIN_SURROGATE, Character.MAX_SURROGATE)
63              .invalid(0xfdd0, 0xfdef)
64              .invalid(0xfffe, 0xffff)
65              .invalid(0x1fffe, 0x1ffff)
66              .invalid(0x2fffe, 0x2ffff)
67              .invalid(0x3fffe, 0x3ffff)
68              .invalid(0x4fffe, 0x4ffff)
69              .invalid(0x5fffe, 0x5ffff)
70              .invalid(0x6fffe, 0x6ffff)
71              .invalid(0x7fffe, 0x7ffff)
72              .invalid(0x8fffe, 0x8ffff)
73              .invalid(0x9fffe, 0x9ffff)
74              .invalid(0xafffe, 0xaffff)
75              .invalid(0xbfffe, 0xbffff)
76              .invalid(0xcfffe, 0xcffff)
77              .invalid(0xdfffe, 0xdffff)
78              .invalid(0xefffe, 0xeffff)
79              .invalid(0xffffe, 0xfffff)
80              .invalid(0x10fffe, 0x10ffff)
81  
82              .validSuite()
83              .invalidSuite(XMLEncoder.INVALID_CHARACTER_REPLACEMENT)
84  
85              .build();
86      }
87  
88      public void testMaxEncodedLength() {
89          CDATAEncoder encoder = new CDATAEncoder();
90          assertEquals(0, encoder.maxEncodedLength(0));
91          assertEquals(1, encoder.maxEncodedLength(1));
92          assertEquals(2, encoder.maxEncodedLength(2));
93          assertEquals(15, encoder.maxEncodedLength(3));
94          assertEquals(16, encoder.maxEncodedLength(4));
95          assertEquals(17, encoder.maxEncodedLength(5));
96          assertEquals(30, encoder.maxEncodedLength(6));
97      }
