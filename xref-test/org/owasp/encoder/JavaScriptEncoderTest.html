35  package org.owasp.encoder;
36  
37  import junit.framework.Test;
38  import junit.framework.TestCase;
39  import junit.framework.TestSuite;
40  import org.owasp.encoder.JavaScriptEncoder.Mode;
41  
42  /**
43   * JavaScriptEncoderTest -- test suite for the JavaScriptEncoder.
44   *
45   * @author Jeff Ichnowski
46   */
47  public class JavaScriptEncoderTest extends TestCase {
48      public static Test suite() {
49          TestSuite suite = new TestSuite();
50          for (int asciiOnly = 0 ; asciiOnly <= 1 ; ++asciiOnly) {
51              for (JavaScriptEncoder.Mode mode : JavaScriptEncoder.Mode.values()) {
52  //                if (!(mode == JavaScriptEncoder.Mode.HTML_CONTENT && asciiOnly == 0)) continue;
53                  EncoderTestSuiteBuilder builder = new EncoderTestSuiteBuilder(new JavaScriptEncoder(mode, asciiOnly==1), "(safe)", "(\\)")
54                      .encoded(0, 0x1f)
55                      .valid(' ', '~')
56                      .encoded("\\\'\"");
57  
58                  switch (mode) {
59                  case SOURCE:
60                  case BLOCK:
61                      builder
62                          .encode("\\\"", "\"")
63                          .encode("\\\'", "\'");
64                      break;
65                  case HTML:
66                  case ATTRIBUTE:
67                      builder
68                          .encode("\\x22", "\"")
69                          .encode("\\x27", "\'");
70                      break;
71                  default:
72                      throw new AssertionError("unexpected mode: "+mode);
73                  }
74  
75                  switch (mode) {
76                  case BLOCK:
77                  case HTML:
78                      builder
79                          .encode("\\/", "/")
80                          .encode("\\-", "-")
81                          .encoded("/-");
82                      break;
83                  default:
84                      builder.encode("/", "/");
85                      break;
86                  }
87                  if (mode != Mode.SOURCE) {
88                      builder.encoded("&");
89                  }
90  
91                  builder
92                      .encode("\\\\", "\\")
93                      .encode("backspace", "\\b", "\b")
94                      .encode("tab", "\\t", "\t")
95                      .encode("LF", "\\n", "\n")
96                      .encode("vtab", "\\x0b", "\u000b")
97                      .encode("FF", "\\f", "\f")
98                      .encode("CR", "\\r", "\r")
99                      .encode("NUL", "\\x00", "\0")
100                     .encode("Line Separator", "\\u2028", "\u2028")
101                     .encode("Paragraph Separator", "\\u2029", "\u2029")
102                     .encode("abc", "abc")
103                     .encode("ABC", "ABC");
104 
105                 if (asciiOnly == 0) {
106                     builder
107                         .encode("unicode", "\u1234", "\u1234")
108                         .encode("high-ascii", "\u00ff", "\u00ff")
109                         .valid(0x7f, Character.MAX_CODE_POINT)
110                         .encoded("\u2028\u2029");
111                 } else {
112                     builder
113                         .encode("unicode", "\\u1234", "\u1234")
114                         .encode("high-ascii", "\\xff", "\u00ff")
115                         .encoded(0x7f, Character.MAX_CODE_POINT);
116                 }
117 
118                 suite.addTest(builder
119                     .validSuite()
120                     .encodedSuite()
121                     .build());
122             }
123         }
124         return suite;
125     }
126 }
