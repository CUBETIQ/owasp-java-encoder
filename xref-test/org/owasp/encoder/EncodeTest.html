37  import java.io.IOException;
38  import java.io.StringWriter;
39  import junit.framework.Test;

43  /**
44   * EncodeTest -- exercises the core encode loops in {@link Encode.Buffer}.
45   *
46   * @author Jeff Ichnowski
47   */
48  public class EncodeTest extends TestCase {
49      public static Test suite() {
50          return new TestSuite(EncodeTest.class);
51      }
52  
53      public void testEncodeToWriter() throws IOException {
54          StringWriter out = new StringWriter();
55          Encode.forXml(out, "<script>");
56          assertEquals("&lt;script&gt;", out.toString());
57      }
58  
59      /**
60       * This test validates the logic for dealing with large encodes that
61       * require multiple flushes to the writer because they exceed the
62       * internal buffer size of the encode loop.
63       *
64       * @throws IOException should not be thrown
65       */
66      public void testBatchedEncodingToWriter() throws IOException {
67          String input = "&&&&&&&&&&&&&&&&&&&&"
68              .replace("&", "&&&&&&&&&&&&&&&&&&&&")  // 400
69              .replace("&", "&&&&&&&&&&&&&&&&&&&&"); // 8000
70          String expected = input.replace("&", "&amp;");
71          final int[] appendCount = new int[1];
72          StringWriter out = new StringWriter() {
73              @Override
74              public void write(char[] cbuf, int off, int len) {
75                  super.write(cbuf, off, len);
76                  appendCount[0]++;
77              }
78          };
79          Encode.forXml(out, input);
80  
81          // this assert here is validating that we are encoding with output
82          // that is larger than the internal buffers of the encode loop.
83          // the only way this should fail is if the buffer in the encode loop
84          // is made larger without also updating this test.
85          assertTrue(appendCount[0] > 1);
86  
87          assertEquals(8000 * 5, out.getBuffer().length());
88          assertEquals(expected, out.toString());
89      }
90  
91      /**
92       * This method tests that strings that do not require encoding are passed
93       * directly to the writer as a string, without any additional buffer
94       * copies.
95       *
96       * @throws IOException not thrown
97       */
98      public void testUnencodedStringToWriter() throws IOException {
99          final String unencodedString = "safe";
100         final boolean[] called = new boolean[1];
101         StringWriter out = new StringWriter() {
102             @Override
103             public void write(String str) {
104                 called[0] = true;
105                 assertSame(unencodedString, str);
106                 super.write(str);
107             }
108         };
109         Encode.forXml(out, unencodedString);
110         assertTrue(called[0]);
111         assertEquals(unencodedString, out.toString());
112     }
113 
114     public void testEncodeNullToWriter() throws IOException {
115         StringWriter out = new StringWriter();
116         Encode.forXml(out, null);
117         assertEquals("null", out.toString());
118     }
119 
120     public void testEncodeString() {
121         assertEquals("&lt;script&gt;", Encode.forXml("<script>"));
122     }
123 
124     public void testEncodeNullToString() {
125         assertEquals("null", Encode.forXml(null));
126     }
127 
128     /**
129      * Make sure return the argument when the string is not encoded.
130      */
131     public void testUnencodedString() {
132         String input = "safe";
133         assertSame(input, Encode.forXml(input));
134     }
135 
136     public void testLargeEncodeToString() {
137         final String input = "&&&&&&&&&&"
138             .replace("&", "&&&&&&&&&&") // 100
139             .replace("&", "&&&&&&&&&&"); // 1000
140 
141         // this should be smaller than the internal input buffer
142         // but cause overflow to the output buffer
143         assertEquals(1000, input.length());
144         String output = Encode.forXml(input);
145         assertEquals(5000, output.length());
146         assertEquals(input.replace("&", "&amp;"), output);
147     }
148 
149     public void testVeryLargeEncodeToString() {
150         final String input = "&&&&&&&&&&&&&&&&&&&&"
151             .replace("&", "&&&&&&&&&&&&&&&&&&&&") // 400
152             .replace("&", "&&&&&&&&&&&&&&&&&&&&"); // 8000
153 
154         // this should be larger than the internal input buffer
155         assertEquals(8000, input.length());
156         String output = Encode.forXml(input);
157         assertEquals(40000, output.length());
158         assertEquals(input.replace("&", "&amp;"), output);
159     }
