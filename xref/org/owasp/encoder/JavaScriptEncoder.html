34  package org.owasp.encoder;
35  
36  import java.nio.CharBuffer;
37  import java.nio.charset.CoderResult;
38  
39  /**
40   * JavaScriptEncoder -- An encoder for JavaScript string contexts.
41   *
42   * @author jeffi
43   */
44  class JavaScriptEncoder extends Encoder {
45  
46      /**
47       * Mode of operation constants for the JavaScriptEncoder.
48       */
49      enum Mode {
50          /**
51           * Standard encoding of JavaScript Strings. Escape sequences are chosen
52           * according to what is the shortest sequence possible for the
53           * character.
54           */
55          SOURCE,
56          /**
57           * Encoding for use in HTML attributes. Quote characters are escaped
58           * using hex encodes instead of backslashes. The alternate would be to
59           * use a sequence of encodes that would actually be longer. In this mode
60           * double-quote is "\x22" and single-quote is "\x27". (In HTML
61           * attributes the alternate would be encoding "\"" and "\'" with entity
62           * escapes to "\&amp;#34;" and "\&amp;39;").
63           */
64          ATTRIBUTE,
65          /**
66           * Encoding for use in HTML script blocks. The main concern here is
67           * permaturely terminating a script block with a closing "&lt;/" inside
68           * the string. This encoding escapes "/" as "\/" to prevent such
69           * termination.
70           */
71          BLOCK,
72          /**
73           * Encodes for use in either HTML script attributes or blocks.
74           * Essentially this is both special escapes from HTML_ATTRIBUTE and
75           * HTML_CONTENT combined.
76           */
77          HTML,;
78      }
79  
80      /**
81       * The mode of operations--used for toString implementation.
82       */
83      private final Mode _mode;
84      /**
85       * True if quotation characters should be hex encoded. Hex encoding quotes
86       * allows JavaScript to be included in XML attributes without additional
87       * XML-based encoding.
88       */
89      private final boolean _hexEncodeQuotes;
90      /**
91       * An array of 4 32-bit integers used as bitmasks to check if a character
92       * needs encoding or not. If the bit is set, the character is valid and does
93       * not need encoding.
94       */
95      private final int[] _validMasks;
96      /**
97       * True if the output should only include ASCII characters. Valid non-ASCII
98       * characters that would normally not be encoded, will be encoded.
99       */
100     private final boolean _asciiOnly;
101 
102     /**
103      * Constructs a new JavaScriptEncoder for the specified contextual mode.
104      *
105      * @param mode the mode of operation
106      * @param asciiOnly true if only ASCII characters should be included in the
107      * output (all code-points outside the ASCII range will be encoded).
108      */
109     JavaScriptEncoder(Mode mode, boolean asciiOnly) {
110         // TODO: after some testing it appears that an array of int masks
111         // is faster than two longs, or an array of longs or an array of bytes
112         // the other encoders based upon masks should be switched to ints.
113         // (to be clear, it's much faster on 32-bit VMS, and just slightly
114         // faster on 64-bit VMS)
115         _mode = mode;
116 
117         // Note: this probably needs to be repeated everywhere this trick is
118         // used, but here seems like as good a place as any.  According to
119         // the Java spec (x << y) where x and y are integers, is evaluated
120         // as (x << (y & 31)).  Or put another way, only the lower 5 bits
121         // of the shift amount are considered.
122         _validMasks = new int[]{
123             0,
124             -1 & ~((1 << '\'') | (1 << '\"')),
125             -1 & ~((1 << '\\')),
126             asciiOnly ? ~(1 << Unicode.DEL) : -1,};
127 
128         if (mode == Mode.BLOCK || mode == Mode.HTML) {
129             // in <script> blocks, we need to prevent the browser from seeing
130             // "</anything>" and "<!--". To do so we escape "/" as "\/" and
131             // escape "-" as "\-".  Both could be solved with a hex encoding
132             // on "<" but we figure "<" appears often in script strings and
133             // the backslash encoding is more readable than a hex encoding.
134             // (And note, a backslash encoding would not prevent the exploits
135             // on "</...>" and "<!--".
136             // In short "</script>" is escaped as "<\/script>" and "<!--" is
137             // escaped as "<!\-\-".
138             _validMasks[1] &= ~((1 << '/') | (1 << '-'));
139         }
140         if (mode != Mode.SOURCE) {
141             _validMasks[1] &= ~(1 << '&');
142         }
143 
144         _asciiOnly = asciiOnly;
145         _hexEncodeQuotes = (mode == Mode.ATTRIBUTE || mode == Mode.HTML);
146     }
147 
148     @Override
149     int maxEncodedLength(int n) {
150         return n * 6;
151     }
152 
153     @Override
154     int firstEncodedOffset(String input, int off, int len) {
155         final int n = off + len;
156         final int[] validMasks = this._validMasks;
157         for (int i = off; i < n; ++i) {
158             char ch = input.charAt(i);
159             if (ch < 128) {
160                 if ((validMasks[ch >>> 5] & (1 << ch)) == 0) {
161                     return i;
162                 }
163             } else if (_asciiOnly || ch == Unicode.LINE_SEPARATOR || ch == Unicode.PARAGRAPH_SEPARATOR) {
164                 return i;
165             }
166         }
167         return n;
168     }
169 
170     @Override
171     protected CoderResult encodeArrays(CharBuffer input, CharBuffer output, boolean endOfInput) {
172         final char[] in = input.array();
173         final char[] out = output.array();
174         int i = input.arrayOffset() + input.position();
175         final int n = input.arrayOffset() + input.limit();
176         int j = output.arrayOffset() + output.position();
177         final int m = output.arrayOffset() + output.limit();
178 
179         final int[] validMasks = this._validMasks;
180 
181         for (; i < n; ++i) {
182             char ch = in[i];
183 
184             hexEncoded:
185             {
186                 encoded:
187                 {
188                     if (ch < 128) {
189                         if ((validMasks[ch >>> 5] & (1 << ch)) == 0) {
190                             break encoded;
191                         }
192                     } else if (_asciiOnly || ch == Unicode.LINE_SEPARATOR || ch == Unicode.PARAGRAPH_SEPARATOR) {
193                         if (ch <= 0xff) {
194                             break hexEncoded;
195                         }
196                         if (j + 6 > m) {
197                             return overflow(input, i, output, j);
198                         }
199                         out[j++] = '\\';
200                         out[j++] = 'u';
201                         out[j++] = HEX[ch >>> 3 * HEX_SHIFT];
202                         out[j++] = HEX[ch >>> 2 * HEX_SHIFT & HEX_MASK];
203                         out[j++] = HEX[ch >>> HEX_SHIFT & HEX_MASK];
204                         out[j++] = HEX[ch & HEX_MASK];
205                         continue;
206                     }
207                     if (j >= m) {
208                         return overflow(input, i, output, j);
209                     }
210                     out[j++] = ch;
211                     continue;
212                 }
213 
214                 switch (ch) {
215                     case '\b':
216                         if (j + 2 > m) {
217                             return overflow(input, i, output, j);
218                         }
219                         out[j++] = '\\';
220                         out[j++] = 'b';
221                         continue;
222                     case '\t':
223                         if (j + 2 > m) {
224                             return overflow(input, i, output, j);
225                         }
226                         out[j++] = '\\';
227                         out[j++] = 't';
228                         continue;
229                     case '\n':
230                         if (j + 2 > m) {
231                             return overflow(input, i, output, j);
232                         }
233                         out[j++] = '\\';
234                         out[j++] = 'n';
235                         continue;
236                     // Per Mike Samuel "\v should not be used since some
237                     // versions of IE treat it as a literal letter 'v'"
238 //                case 0x0b: // '\v'
239 //                    if (j+1 >= m) {
240 //                        return overflow(input, i, output, j);
241 //                    }
242 //                    out[j++] = '\\';
243 //                    out[j++] = 'v';
244 //                    break;
245                     case '\f':
246                         if (j + 2 > m) {
247                             return overflow(input, i, output, j);
248                         }
249                         out[j++] = '\\';
250                         out[j++] = 'f';
251                         continue;
252                     case '\r':
253                         if (j + 2 > m) {
254                             return overflow(input, i, output, j);
255                         }
256                         out[j++] = '\\';
257                         out[j++] = 'r';
258                         continue;
259                     case '\'':
260                     case '\"':
261                         if (_hexEncodeQuotes) {
262                             break hexEncoded;
263                         }
264                     // fall through
265                     case '\\':
266                     case '/':
267                     case '-':
268                         // We'll only see '/' and '-' here in the BLOCK and HTML
269                         // modes otherwise it will be accepted as valid by the
270                         // bitmasks.
271                         if (j + 2 > m) {
272                             return overflow(input, i, output, j);
273                         }
274                         out[j++] = '\\';
275                         out[j++] = ch;
276                         continue;
277                     default:
278                         break;
279                 }
280             }
281 
282             if (j + 4 > m) {
283                 return overflow(input, i, output, j);
284             }
285             out[j++] = '\\';
286             out[j++] = 'x';
287             out[j++] = HEX[ch >>> HEX_SHIFT];
288             out[j++] = HEX[ch & HEX_MASK];
289         }
290 
291         return underflow(input, i, output, j);
292     }
293 
294     @Override
295     public String toString() {
296         return "JavaScriptEncoder(mode=" + _mode + "," + (_asciiOnly ? "ASCII" : "UNICODE") + ")";
297     }
298 }
