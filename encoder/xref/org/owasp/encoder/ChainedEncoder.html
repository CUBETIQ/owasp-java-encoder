35  package org.owasp.encoder;
36  
37  import java.nio.CharBuffer;
38  import java.nio.charset.CoderResult;
39  
40  /**
41   * ChainedEncoder -- An encoder that chains together two encoders in
42   * order.  This is included as an example, but not actually exposed or
43   * used as it requires an internal buffer making it not thread-safe.
44   * Sequences of 3 or more encodings require chaining together chained
45   * encoders.
46   *
47   * @author Jeff Ichnowski
48   */
49  class ChainedEncoder extends Encoder {
50      /** The first encoder to apply in sequence. */
51      final Encoder _first;
52      /** The second encoder to apply in sequence. */
53      final Encoder _last;
54  
55      /**
56       * The buffer used to store the output of the first encoder before
57       * sending as input to the second encoder.
58       */
59      // TODO: because of this buffer, the ChainedEncoder is the only stateful
60      // encoder.  This needs to be removed somehow, or state control APIs need
61      // to be added (E.g. reset/flush)
62      final CharBuffer _buffer = CharBuffer.allocate(1024);
63  
64      /**
65       * Creates an ChainedEncoder that applies the encoding sequence
66       * {@code input --> first --> last --> output}.
67       *
68       * @param first the first encoder to apply
69       * @param last the second/last encoder to apply.
70       */
71      ChainedEncoder(Encoder first, Encoder last) {
72          _first = first;
73          _last = last;
74      }
75  
76      /**
77       * Encodes an input string to an output string.
78       *
79       * @param str the string to encode
80       * @return the encoded string.
81       */
82      public String encode(String str) {
83          if (str == null) {
84              str = "null";
85          }
86  
87          int n = str.length();
88          int j = _first.firstEncodedOffset(str, 0, n);
89          if (j == n) {
90              // string is unchanged after encoding with the first encoder
91              return Encode.encode(_last, str);
92          }
93  
94          final int remaining = n - j;
95          final int m = j + _first.maxEncodedLength(n);
96          CharBuffer input = CharBuffer.allocate(m);
97          str.getChars(0, j, input.array(), 0);
98          str.getChars(j, n, input.array(), m - remaining);
99  
100         input.limit(m).position(m - remaining);
101         CharBuffer tmp = input.duplicate();
102         tmp.position(j);
103 
104         CoderResult cr = _first.encode(input, tmp, true);
105         assert cr.isUnderflow() : "maxEncodedLength was incorrect";
106 
107         CharBuffer output = CharBuffer.allocate(_last.maxEncodedLength(tmp.position()));
108         tmp.flip();
109 
110         cr = _last.encode(tmp, output, true);
111         assert cr.isUnderflow() : "maxEncodedLength was incorrect";
112 
113         return new String(output.array(), 0, output.position());
114     }
115 
116     @Override
117     protected int firstEncodedOffset(String input, int off, int len) {
118         int i = _first.firstEncodedOffset(input, off, len);
119         return _last.firstEncodedOffset(input, off, i - off);
120     }
121 
122     @Override
123     protected int maxEncodedLength(int n) {
124         return _last.maxEncodedLength(
125             _first.maxEncodedLength(n));
126     }

135     @Override
136     public CoderResult encode(CharBuffer input, CharBuffer output, boolean endOfInput) {
137         for (;;) {
138             CoderResult cr1 = _first.encode(input, _buffer, endOfInput);
139             _buffer.flip();
140             CoderResult cr2 = _last.encode(_buffer, output, endOfInput && cr1.isUnderflow());
141             _buffer.compact();
142 
143             if (cr2.isOverflow()) {
144                 // no more room in output, even if we still have buffered input.
145                 return cr2;
146             }
147 
148             if (cr1.isUnderflow()) {
149                 // we've consumed everything from input
150                 return cr2;
151             }
152         }
153     }
154 
155     @Override
156     protected CoderResult encodeArrays(CharBuffer input, CharBuffer output, boolean endOfInput) {
157         return encode(input, output, endOfInput); // TODO: invert.
158     }
159 
160     @Override
161     public String toString() {
162         return "["+_first+","+_last+"]";
163     }
164 }
