35  package org.owasp.encoder;
36  
37  import junit.framework.Test;
38  import junit.framework.TestCase;
39  
40  /**
41   * JavaEncoderTest -- test suite for the JavaEncoder.
42   *
43   * @author Jeff Ichnowski
44   */
45  public class JavaEncoderTest extends TestCase {
46      public static Test suite() {
47          return new EncoderTestSuiteBuilder(new JavaEncoder(), "-safe-", "'")
48              .encode("\\'", "\'")
49              .encode("\\\"", "\"")
50              .encode("\\\\", "\\")
51              .encode("abc123XYZ", "abc123XYZ")
52              .encode("\\b", "\\b", "\b")
53              .encode("\\t", "\\t", "\t")
54              .encode("\\n", "\\n", "\n")
55              .encode("\\f", "\\f", "\f")
56              .encode("\\r", "\\r", "\r")
57              .encode("\\0", "\\0", "\0")
58              .encode("\\17", "\\17", "\17")
59              .encode("\\377", "\\377", "\377")
60              .encode("\\u1234", "\\u1234", "\u1234")
61              .encode("\\uabcd", "\\uabcd", "\uabcd")
62              .encode("invalid-surrogate", "\\udc00", "\udc00") // are okay
63              .encode("\\0\\0", "\\0\\0", "\0\0")
64              .encode("\\0 + '0'", "\\0000", "\0000")
65              .encode("\\37 + '0'", "\\0370", "\0370")
66  
67              .encoded(0, Character.MAX_CODE_POINT)
68              .valid(' ', '~')
69              .encoded("\"\'\\")
70  
71              .validSuite()
72              .encodedSuite()
73  
74              .build();
75      }
76  }
